FROM

👉 Defines base image

FROM python:3.10

WORKDIR

👉 Sets working directory

WORKDIR /app
COPY

👉 Copies files to container

COPY . .

ADD

👉 Similar to COPY (extra features like URL, tar)

ADD file.tar /app/
RUN

👉 Executes commands during build

RUN pip install -r requirements.txt
CMD

👉 Default command to run container

CMD ["python", "app.py"]

ENTRYPOINT

👉 Main executable

ENTRYPOINT ["python"]

EXPOSE

👉 Defines port

EXPOSE 5000
